1182class Mock(CallableMixin, NonCallableMock):

1184 Create a new `Mock` object. `Mock` takes several optional arguments 

1185 that specify the behaviour of the Mock object: 

1186 

1187 * `spec`: This can be either a list of strings or an existing object (a 

1188 class or instance) that acts as the specification for the mock object. If 

1189 you pass in an object then a list of strings is formed by calling dir on 

1190 the object (excluding unsupported magic attributes and methods). Accessing 

1191 any attribute not in this list will raise an `AttributeError`. 

1192 

1193 If `spec` is an object (rather than a list of strings) then 

1194 `mock.__class__` returns the class of the spec object. This allows mocks 

1195 to pass `isinstance` tests. 

1196 

1197 * `spec_set`: A stricter variant of `spec`. If used, attempting to *set* 

1198 or get an attribute on the mock that isn't on the object passed as 

1199 `spec_set` will raise an `AttributeError`. 

1200 

1201 * `side_effect`: A function to be called whenever the Mock is called. See 

1202 the `side_effect` attribute. Useful for raising exceptions or 

1203 dynamically changing return values. The function is called with the same 

1204 arguments as the mock, and unless it returns `DEFAULT`, the return 

1205 value of this function is used as the return value. 

1206 

1207 If `side_effect` is an iterable then each call to the mock will return 

1208 the next value from the iterable. If any of the members of the iterable 

1209 are exceptions they will be raised instead of returned. 

1210

1212 this is a new Mock (created on first access). See the 

1213 `return_value` attribute. 

1214

1216 calling the Mock will pass the call through to the wrapped object 

1217 (returning the real result). Attribute access on the mock will return a 

1218 Mock object that wraps the corresponding attribute of the wrapped object 

1219 (so attempting to access an attribute that doesn't exist will raise an 

1220 `AttributeError`). 

1221

1223 to the wrapped object and the `return_value` is returned instead.

1226 mock. This can be useful for debugging. The name is propagated to child

1393 def __enter__(self): 

1394 """Perform the patch.""" 

1395 new, spec, spec_set = self.new, self.spec, self.spec_set 

1396 autospec, kwargs = self.autospec, self.kwargs 

1397 new_callable = self.new_callable 

1398 self.target = self.getter() 

1399 

1400 # normalise False to None 

1401 if spec is False: 

1402 spec = None 

1403 if spec_set is False: 

1404 spec_set = None 

1405 if autospec is False: 

1406 autospec = None 

1407 

1408 if spec is not None and autospec is not None: 

1409 raise TypeError("Can't specify spec and autospec") 

1410 if ((spec is not None or autospec is not None) and 

1411 spec_set not in (True, None)): 

1412 raise TypeError("Can't provide explicit spec_set *and* spec or autospec") 

1413 

1414 original, local = self.get_original() 

1415 

1416 if new is DEFAULT and autospec is None: 

1417 inherit = False 

1418 if spec is True: 

1419 # set spec to the object we are replacing 

1420 spec = original 

1421 if spec_set is True: 

1422 spec_set = original 

1423 spec = None 

1424 elif spec is not None: 

1425 if spec_set is True: 

1426 spec_set = spec 

1427 spec = None 

1428 elif spec_set is True: 

1429 spec_set = original 

1430 

1431 if spec is not None or spec_set is not None: 

1432 if original is DEFAULT: 

1433 raise TypeError("Can't use 'spec' with create=True") 

1434 if isinstance(original, type): 

1435 # If we're patching out a class and there is a spec 

1436 inherit = True 

1437 if spec is None and _is_async_obj(original): 

1438 Klass = AsyncMock 

1439 else: 

1440 Klass = MagicMock 

1441 _kwargs = {} 

1442 if new_callable is not None: 

1443 Klass = new_callable 

1444 elif spec is not None or spec_set is not None: 

1445 this_spec = spec 

1446 if spec_set is not None: 

1447 this_spec = spec_set 

1448 if _is_list(this_spec): 

1449 not_callable = '__call__' not in this_spec 

1450 else: 

1451 not_callable = not callable(this_spec) 

1452 if _is_async_obj(this_spec): 

1453 Klass = AsyncMock 

1454 elif not_callable: 

1455 Klass = NonCallableMagicMock 

1456 

1457 if spec is not None: 

1458 _kwargs['spec'] = spec 

1459 if spec_set is not None: 

1460 _kwargs['spec_set'] = spec_set 

1461 

1462 # add a name to mocks 

1463 if (isinstance(Klass, type) and 

1464 issubclass(Klass, NonCallableMock) and self.attribute): 

1465 _kwargs['name'] = self.attribute 

1466 

1467 _kwargs.update(kwargs) 

1468 new = Klass(**_kwargs) 

1469 

1470 if inherit and _is_instance_mock(new): 

1471 # we can only tell if the instance should be callable if the 

1472 # spec is not a list 

1473 this_spec = spec 

1474 if spec_set is not None: 

1475 this_spec = spec_set 

1476 if (not _is_list(this_spec) and not 

1477 _instance_callable(this_spec)): 

1478 Klass = NonCallableMagicMock 

1479 

1480 _kwargs.pop('name') 

1481 new.return_value = Klass(_new_parent=new, _new_name='()', 

1482 **_kwargs) 

1483 elif autospec is not None: 

1484 # spec is ignored, new *must* be default, spec_set is treated 

1485 # as a boolean. Should we check spec is not None and that spec_set 

1486 # is a bool? 

1487 if new is not DEFAULT: 

1488 raise TypeError( 

1489 "autospec creates the mock for you. Can't specify " 

1490 "autospec and new." 

1491 ) 

1492 if original is DEFAULT: 

1493 raise TypeError("Can't use 'autospec' with create=True") 

1494 spec_set = bool(spec_set) 

1495 if autospec is True: 

1496 autospec = original 

1497 

1498 new = create_autospec(autospec, spec_set=spec_set, 

1499 _name=self.attribute, **kwargs) 

1500 elif kwargs: 

1501 # can't set keyword args when we aren't creating the mock 

1502 # XXXX If new is a Mock we could call new.configure_mock(**kwargs) 

1503 raise TypeError("Can't pass kwargs to a mock we aren't creating") 

1504 

1505 new_attr = new 

1506 

1507 self.temp_original = original 

1508 self.is_local = local 

1509 self._exit_stack = contextlib.ExitStack() 

1510 try: 

1511 setattr(self.target, self.attribute, new_attr) 

1512 if self.attribute_name is not None: 

1513 extra_args = {} 

1514 if self.new is DEFAULT: 

1515 extra_args[self.attribute_name] = new 

1516 for patching in self.additional_patchers: 

1517 arg = self._exit_stack.enter_context(patching) 

1518 if patching.new is DEFAULT: 

1519 extra_args.update(arg) 

1520 return extra_args 

1521 

1522 return new 

1523 except: 

1524 if not self.__exit__(*sys.exc_info()): 

1525 raise

1527 def __exit__(self, *exc_info): 

1528 """Undo the patch.""" 

1529 if self.is_local and self.temp_original is not DEFAULT: 

1530 setattr(self.target, self.attribute, self.temp_original) 

1531 else: 

1532 delattr(self.target, self.attribute) 

1533 if not self.create and (not hasattr(self.target, self.attribute) or 

1534 self.attribute in ('__doc__', '__module__', 

1535 '__defaults__', '__annotations__', 

1536 '__kwdefaults__')): 

1537 # needed for proxy objects like django settings 

1538 setattr(self.target, self.attribute, self.temp_original) 

1539 

1540 del self.temp_original 

1541 del self.is_local 

1542 del self.target 

1543 exit_stack = self._exit_stack 

1544 del self._exit_stack 

1545 return exit_stack.__exit__(*exc_info)

1655def patch( 

1656 target, new=DEFAULT, spec=None, create=False, 

1657 spec_set=None, autospec=None, new_callable=None, **kwargs 

1658 ): 

1659 """ 

1660 `patch` acts as a function decorator, class decorator or a context 

1661 manager. Inside the body of the function or with statement, the `target` 

1662 is patched with a `new` object. When the function/with statement exits 

1663 the patch is undone. 

1664 

1665 If `new` is omitted, then the target is replaced with an 

1666 `AsyncMock if the patched object is an async function or a 

1667 `MagicMock` otherwise. If `patch` is used as a decorator and `new` is 

1668 omitted, the created mock is passed in as an extra argument to the 

1669 decorated function. If `patch` is used as a context manager the created 

1670 mock is returned by the context manager. 

1671 

1672 `target` should be a string in the form `'package.module.ClassName'`. The 

1673 `target` is imported and the specified object replaced with the `new` 

1674 object, so the `target` must be importable from the environment you are 

1675 calling `patch` from. The target is imported when the decorated function 

1676 is executed, not at decoration time. 

1677 

1678 The `spec` and `spec_set` keyword arguments are passed to the `MagicMock` 

1679 if patch is creating one for you. 

1680 

1681 In addition you can pass `spec=True` or `spec_set=True`, which causes 

1682 patch to pass in the object being mocked as the spec/spec_set object. 

1683 

1684 `new_callable` allows you to specify a different class, or callable object, 

1685 that will be called to create the `new` object. By default `AsyncMock` is 

1686 used for async functions and `MagicMock` for the rest. 

1687 

1688 A more powerful form of `spec` is `autospec`. If you set `autospec=True` 

1689 then the mock will be created with a spec from the object being replaced. 

1690 All attributes of the mock will also have the spec of the corresponding 

1691 attribute of the object being replaced. Methods and functions being 

1692 mocked will have their arguments checked and will raise a `TypeError` if 

1693 they are called with the wrong signature. For mocks replacing a class, 

1694 their return value (the 'instance') will have the same spec as the class. 

1695 

1696 Instead of `autospec=True` you can pass `autospec=some_object` to use an 

1697 arbitrary object as the spec instead of the one being replaced. 

1698 

1699 By default `patch` will fail to replace attributes that don't exist. If 

1700 you pass in `create=True`, and the attribute doesn't exist, patch will 

1701 create the attribute for you when the patched function is called, and 

1702 delete it again afterwards. This is useful for writing tests against 

1703 attributes that your production code creates at runtime. It is off by 

1704 default because it can be dangerous. With it switched on you can write 

1705 passing tests against APIs that don't actually exist! 

1706 

1707 Patch can be used as a `TestCase` class decorator. It works by 

1708 decorating each test method in the class. This reduces the boilerplate 

1709 code when your test methods share a common patchings set. `patch` finds 

1710 tests by looking for method names that start with `patch.TEST_PREFIX`. 

1711 By default this is `test`, which matches the way `unittest` finds tests. 

1712 You can specify an alternative prefix by setting `patch.TEST_PREFIX`. 

1713 

1714 Patch can be used as a context manager, with the with statement. Here the 

1715 patching applies to the indented block after the with statement. If you 

1716 use "as" then the patched object will be bound to the name after the 

1717 "as"; very useful if `patch` is creating a mock object for you. 

1718 

1719 `patch` takes arbitrary keyword arguments. These will be passed to 

1720 `AsyncMock` if the patched object is asynchronous, to `MagicMock` 

1721 otherwise or to `new_callable` if specified. 

1722 

1723 `patch.dict(...)`, `patch.multiple(...)` and `patch.object(...)` are 

1724 available for alternate use-cases. 

1725 """ 

1726 getter, attribute = _get_target(target) 

1727 return _patch( 

1728 getter, attribute, new, spec, create, 

1729 spec_set, autospec, new_callable, kwargs 

1730 )

1733class _patch_dict(object): 

1734 """ 

1735 Patch a dictionary, or dictionary like object, and restore the dictionary 

1736 to its original state after the test. 

1737 

1738 `in_dict` can be a dictionary or a mapping like container. If it is a 

1739 mapping then it must at least support getting, setting and deleting items 

1740 plus iterating over keys. 

1741 

1742 `in_dict` can also be a string specifying the name of the dictionary, which 

1743 will then be fetched by importing it. 

1744 

1745 `values` can be a dictionary of values to set in the dictionary. `values` 

1746 can also be an iterable of `(key, value)` pairs. 

1747 

1748 If `clear` is True then the dictionary will be cleared before the new 

1749 values are set. 

1750 

1751 `patch.dict` can also be called with arbitrary keyword arguments to set 

1752 values in the dictionary:: 

1753 

1754 with patch.dict('sys.modules', mymodule=Mock(), other_module=Mock()): 

1755 ... 

1756 

1757 `patch.dict` can be used as a context manager, decorator or class 

1758 decorator. When used as a class decorator `patch.dict` honours 

1759 `patch.TEST_PREFIX` for choosing which methods to wrap. 

1760 """ 

1761 

1762 def __init__(self, in_dict, values=(), clear=False, **kwargs): 

1763 self.in_dict = in_dict 

1764 # support any argument supported by dict(...) constructor 

1765 self.values = dict(values) 

1766 self.values.update(kwargs) 

1767 self.clear = clear 

1768 self._original = None 

1769 

1770 

1771 def __call__(self, f): 

1772 if isinstance(f, type): 

1773 return self.decorate_class(f) 

1774 @wraps(f) 

1775 def _inner(*args, **kw): 

1776 self._patch_dict() 

1777 try: 

1778 return f(*args, **kw) 

1779 finally: 

1780 self._unpatch_dict() 

1781 

1782 return _inner

1785 def decorate_class(self, klass): 

1786 for attr in dir(klass): 

1787 attr_value = getattr(klass, attr) 

1788 if (attr.startswith(patch.TEST_PREFIX) and 

1789 hasattr(attr_value, "__call__")): 

1790 decorator = _patch_dict(self.in_dict, self.values, self.clear) 

1791 decorated = decorator(attr_value) 

1792 setattr(klass, attr, decorated) 

1793 return klass
